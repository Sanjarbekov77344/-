## Самостоятельная работа №1

class Tomato:
    states = {
        0: 'отсутствие помидоров',
        1: 'цветение помидоров',
        2: 'зеленый помидор',
        3: 'красный помидор'
    }

    def __init__(self, index):
        self._index = index
        self._state = 0

    def grow(self):
        if self._state < 3:
            self._state += 1
        print(f"Томат {self._index} теперь на стадии: {Tomato.states[self._state]}")

    def is_ripe(self):
        return self._state == 3


class TomatoBush:
    def __init__(self, num_tomatoes):
        self.tomatoes = [Tomato(i) for i in range(num_tomatoes)]

    def grow_all(self):
        for tomato in self.tomatoes:
            tomato.grow()

    def all_are_ripe(self):
        return all(tomato.is_ripe() for tomato in self.tomatoes)

    def give_away_all(self):
        self.tomatoes = []


class Gardener:
    @staticmethod
    def knowledge_base():
        print(" " * 2)
        print("СПРАВКА ПО САДОВОДСТВУ:")
        print("1. Помидор проходит 4 стадии созревания: отсутствие плода, цветение цветка, зеленый (незрелый) помидор, красный (зрелый) помидор ")
        print("2. Садовник должен ухаживать за растением, пока все помидоры не созреют")
        print("3. Собирать урожай можно только когда все помидоры красные")
        print(" " * 2)

    def __init__(self, name, plant):
        self.name = name
        self._plant = plant

    def work(self):
        print(f"{self.name} работает...")
        self._plant.grow_all()

    def harvest(self):
        if self._plant.all_are_ripe():
            print(f"{self.name} собирает урожай! Урожай собран!")
            self._plant.give_away_all()
            return True
        else:
            print(f"{self.name}: Помидоры еще не созрели! Нужно продолжать ухаживать.")
            return False

if __name__ == "__main__":
    Gardener.knowledge_base()
    print()

    bush = TomatoBush(3)
    gardener = Gardener("Даниил", bush)

    print(f"Создан садовник {gardener.name} с кустом из {len(bush.tomatoes)} помидоров")
    print()

    print("Первый день ухода ")
    gardener.work()
    print()

    print("Попытка сбора урожая ")
    gardener.harvest()
    print()

    print("Второй день ухода ")
    gardener.work()
    print()

    print("Третий день ухода ")
    gardener.work()

    else:
        print("Что-то пошло не так...")

    print()
