class CountDown:
    def __init__(self, start):
        self.count = start + 1

    def __iter__(self):
        return self

    def __next__(self):
        self.count -= 1
        if self.count < 0:
            raise StopIteration
        return self.count

if __name__ == '__main__':
    counter = CountDown(5)
    for i in counter:
        print(i)

def fib(n):
    a, b = 1, 1
    count = 0
    while count < n:
        yield a
        a, b = b, a + b
        count += 1

n = 200
fibonacci_sequence = fib(n)
result = list(fibonacci_sequence)[-1]

print(f"200-е число Фибоначчи: {result}")

def fib(n):
    a, b = 1, 1
    count = 0
    while count < n:
        yield a
        a, b = b, a + b
        count += 1

n = 200
fibonacci_sequence = fib(n)

with open('fib.txt', 'w') as file:
    for number in fibonacci_sequence:
        file.write(f"{number}\n")

print("Числа Фибоначчи успешно записаны в файл 'fib.txt'")
